<script>
const STORAGE_KEY = 'exam_v3';

// Subject definitions: id, total topics, which exam
const subjects = [
  { id: 'civil',     total: 19, exams: ['broker','land'] },
  { id: 'trust',     total: 4,  exams: ['land'] },
  { id: 'appraisal', total: 6,  exams: ['broker'] },
  { id: 'landlaw',   total: 8,  exams: ['broker','land'] },
  { id: 'broker',    total: 5,  exams: ['broker'] },
  { id: 'reg',       total: 6,  exams: ['land'] },
  { id: 'chinese',   total: 2,  exams: ['broker'] },
];

function loadState() {
  try { return JSON.parse(localStorage.getItem(STORAGE_KEY)) || {}; } catch(e) { return {}; }
}

function saveState(s) {
  try { localStorage.setItem(STORAGE_KEY, JSON.stringify(s)); } catch(e) {}
}

const STATES = ['', 'current', 'done'];

function toggle(el) {
  if (!el.dataset.id) return;
  const s = loadState();
  const id = el.dataset.id;
  let idx = 0;
  if (el.classList.contains('done')) idx = 2;
  else if (el.classList.contains('current')) idx = 1;
  const next = (idx + 1) % 3;
  el.classList.remove('done', 'current');
  if (STATES[next]) el.classList.add(STATES[next]);
  s[id] = next;
  saveState(s);
  updateAllProgress();
}

function toggleSubject(id) {
  const el = document.getElementById(id);
  if (el) el.classList.toggle('collapsed');
}

function applyState() {
  const s = loadState();
  document.querySelectorAll('.topic-item[data-id]').forEach(el => {
    const id = el.dataset.id;
    if (s[id] !== undefined) {
      el.classList.remove('done', 'current');
      if (STATES[s[id]]) el.classList.add(STATES[s[id]]);
    }
  });
}

function updateAllProgress() {
  let totalDone = 0, totalAll = 0;
  subjects.forEach(subj => {
    const items = document.querySelectorAll(`#subj-${subj.id} .topic-item[data-id]`);
    const done = document.querySelectorAll(`#subj-${subj.id} .topic-item.done`);
    const pct = items.length ? Math.round(done.length / items.length * 100) : 0;
    const bar = document.getElementById(`bar-${subj.id}`);
    const prog = document.getElementById(`prog-${subj.id}`);
    if (bar) bar.style.width = pct + '%';
    if (prog) prog.textContent = done.length + ' / ' + items.length;
    totalDone += done.length;
    totalAll += items.length;
  });
  const overall = totalAll ? Math.round(totalDone / totalAll * 100) : 0;
  document.getElementById('overall-fill').style.width = overall + '%';
  document.getElementById('overall-count').textContent = totalDone + ' / ' + totalAll + '  (' + overall + '%)';
}

function showTab(name) {
  document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
  document.querySelectorAll('.tab-btn').forEach(el => el.classList.remove('active'));
  document.getElementById('tab-' + name).classList.add('active');
  event.target.classList.add('active');

  if (name === 'broker') {
    const container = document.getElementById('broker-subjects');
    container.innerHTML = '';
    ['civil','appraisal','landlaw','broker','chinese'].forEach(id => {
      const clone = document.getElementById('subj-' + id).cloneNode(true);
      clone.id = 'broker-' + id;
      clone.querySelectorAll('.topic-item[data-id]').forEach(item => {
        item.onclick = function() { toggle(document.querySelector(`#subj-${id} [data-id="${item.dataset.id}"]`) || item); };
      });
      container.appendChild(clone);
    });
  }

  if (name === 'land') {
    const container = document.getElementById('land-subjects');
    container.innerHTML = '';
    ['civil','trust','landlaw','reg'].forEach(id => {
      const clone = document.getElementById('subj-' + id).cloneNode(true);
      clone.id = 'land-' + id;
      container.appendChild(clone);
    });
  }
}

function updateCountdown() {
  const now = new Date();
  const d1 = new Date('2026-11-14');
  const d2 = new Date('2027-06-05');
  document.getElementById('days1').textContent = Math.max(0, Math.ceil((d1 - now) / 86400000));
  document.getElementById('days2').textContent = Math.max(0, Math.ceil((d2 - now) / 86400000));
}

applyState();
updateAllProgress();
updateCountdown();
